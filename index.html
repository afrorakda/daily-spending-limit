/* Script部分の修正・追加 */ // ページ読み込み時（または戻ってきた時）に状態を復元する window.addEventListener('pageshow', () => { calculate(); }); function calculate() { budgetInput.value = budgetInput.value.replace(/[^0-9.]/g, ''); daysInput.value = daysInput.value.replace(/[^0-9]/g, ''); const budget = parseFloat(budgetInput.value); const days = parseInt(daysInput.value); const unit = unitInput.value || ''; if (!isNaN(budget) && !isNaN(days) && days > 0) { const daily = budget / days; resultVal.innerText = formatNumber(daily); resUnit.innerText = unit; } else { resultVal.innerText = '0'; resUnit.innerText = ''; } // クリアボタンの状態を強制更新 updateClearButtons(); } function updateClearButtons() { clrBudget.classList.toggle('active', budgetInput.value !== ''); clrDays.classList.toggle('active', daysInput.value !== ''); } function clearInput(id) { document.getElementById(id).value = ''; calculate(); // ここでupdateClearButtonsも呼ばれる document.getElementById(id).focus(); }
